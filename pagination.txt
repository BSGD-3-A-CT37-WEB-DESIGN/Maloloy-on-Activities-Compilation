<script>
let currentPage = 1;
const totalPages = 3;

const pages = {
1: ["gallery/img1.jpg", "gallery/img2.jpg", "gallery/img3.jpg"],
2: ["gallery/img4.jpg", "gallery/img5.jpg", "gallery/img6.jpg"],
3: ["gallery/img7.jpg", "gallery/img8.jpg", "gallery/img9.jpg"]
};

function changePage(pageNum) {
currentPage = pageNum;
const gallery = document.getElementById("gallery");
gallery.innerHTML = ""; 

pages[pageNum].forEach(src => {
const img = document.createElement("img");
img.src = src;
img.alt = "Cat";
gallery.appendChild(img);
});

 
document.querySelectorAll('.page-link').forEach(link => link.classList.remove('active'));
document.getElementById("p" + pageNum).classList.add('active');
  }

 function nextPage() {
    if (currentPage < totalPages) {
      changePage(currentPage + 1);
    }
  }
function prevPage() {
    if (currentPage > 1) {
      changePage(currentPage - 1);
    }
  }
  changePage(1);
</script>
